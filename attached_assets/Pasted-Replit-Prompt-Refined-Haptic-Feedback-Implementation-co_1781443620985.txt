1. Add this Utility Function (recommended location: `utils/haptic.ts` or top of main JS/App file)
// haptic.ts or inline in main file
export const haptic = {
  init() {
    // Ensure Telegram WebApp is ready
    if (window.Telegram?.WebApp) {
      window.Telegram.WebApp.ready();
    }
  },

  trigger(type: 'light' | 'medium' | 'heavy' | 'rigid' | 'soft' | 'selection' | 'success' | 'error' | 'warning' = 'medium') {
    const tg = window.Telegram?.WebApp;
    if (!tg?.HapticFeedback) return;

    try {
      if (['success', 'error', 'warning'].includes(type)) {
        tg.HapticFeedback.notificationOccurred(type as 'success' | 'error' | 'warning');
      } else if (type === 'selection') {
        tg.HapticFeedback.selectionChanged();
      } else {
        tg.HapticFeedback.impactOccurred(type as 'light' | 'medium' | 'heavy' | 'rigid' | 'soft');
      }
    } catch (e) {
      console.warn('Haptic feedback failed:', e);
    }
  }
};
2. Initialize it early (in `main.ts` / `App.tsx` / `index.js`)

4. Apply to All Major Action Buttons
// Global click handler (add once)
document.addEventListener('click', (e) => {
  const target = e.target as HTMLElement;
  if (target.closest('button') || target.closest('.btn') || target.closest('[role="button"]')) {
    // Debounce rapid clicks
    if ((window as any).lastHapticTime && Date.now() - (window as any).lastHapticTime < 80) return;
    
    (window as any).lastHapticTime = Date.now();
    
    const isPrimaryAction = target.closest('.primary-btn, .buy-btn, .confirm-btn');
    haptic.trigger(isPrimaryAction ? 'medium' : 'light');
  }
}, true);
