const [predictions, setPredictions] = useState({});

const generatePrediction = (currentPrice: number, fuelType: string) => {
  haptic.trigger('light');
  
  // Simple "AI" forecast: trend + noise
  const trend = (Math.random() - 0.45) * 0.8; // slight downward bias
  const predicted24h = Math.round((currentPrice * (1 + trend)) * 100) / 100;
  const confidence = 72 + Math.floor(Math.random() * 13);
  
  return {
    current: currentPrice,
    predicted24h,
    trend: predicted24h > currentPrice ? 'up' : 'down',
    confidence,
    lastUpdated: new Date()
  };
};

// Auto-refresh predictions together with prices
useEffect(() => {
  // Call when prices update
  const newPreds = {};
  Object.keys(currentPrices).forEach(key => {

  });
  setPredictions(newPreds);
}, [currentPrices]);
