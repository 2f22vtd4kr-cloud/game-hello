// In price refresh or WebSocket handler
const checkAndShowPriceAlerts = (newPrices: Record) => {
  Object.keys(newPrices).forEach(fuel => {
    const oldP = previousPrices[fuel];
    const newP = newPrices[fuel];
    if (oldP && Math.abs((newP - oldP) / oldP) * 100 >= 1) {
      haptic.trigger('heavy');
      
      showToast({
        type: newP > oldP ? 'warning' : 'success',
        title: `Цена ${fuel} изменилась`,
        message: `${oldP} → ${newP} ₽ (${((newP - oldP)/oldP*100).toFixed(1)}%)`,
        action: {
          label: "Купить сейчас ⛽️",
          onClick: () => switchToTab('catalog')
        },
        duration: 6000
      });
    }
  });
  previousPrices = { ...newPrices };
};
