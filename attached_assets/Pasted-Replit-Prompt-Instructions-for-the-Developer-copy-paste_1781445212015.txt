// Price change detection (in priceService or refresh function)
let previousPrices = {};

const checkPriceChanges = (newPrices: any) => {
  const alerts = [];
  
  Object.keys(newPrices).forEach(fuel => {
    const oldPrice = previousPrices[fuel];
    const newPrice = newPrices[fuel];
    
    if (oldPrice) {
      const change = ((newPrice - oldPrice) / oldPrice) * 100;
      if (Math.abs(change) >= 1) {  // 1%+ change
        alerts.push({
          fuel,
          change: change.toFixed(1),
          direction: change > 0 ? 'up' : 'down',
          old: oldPrice,
          new: newPrice
        });
        
        haptic.trigger(change > 0 ? 'heavy' : 'medium');
      }
    }
  });
  
  if (alerts.length > 0) {
    showPriceAlertBanner(alerts);
  }
  
  previousPrices = { ...newPrices };
};

// Banner component
const showPriceAlertBanner = (alerts: any[]) => {
  // Show temporary toast or banner at top
  const banner = document.createElement('div');
  banner.className = 'fixed top-4 left-1/2 -translate-x-1/2 bg-red-500/90 text-white px-6 py-3 rounded-2xl z-50 flex items-center gap-3';
  banner.innerHTML = `
    ⚠️ Цены изменились!
    ${alerts.map(a => `${a.fuel}: ${a.direction === 'up' ? '↑' : '↓'} ${a.change}%`).join(' • ')}
  `;
  document.body.appendChild(banner);
  
  setTimeout(() => banner.remove(), 5000);
};
